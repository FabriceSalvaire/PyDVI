        url = QtCore.QUrl()
        url.setScheme(Config.Help.url_scheme)
        url.setHost(Config.Help.host)
        url.setPath(Config.Help.url_path_pattern % (Version.pydvi_version.to_string()))

        QtGui.QDesktopServices.openUrl(url)
